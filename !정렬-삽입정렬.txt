#현재 리스트가 거의 정렬된 상태라면 매우 빠르게 동작
#O(N^2)
arr = [1, 7, 3, 4]

for i in range(1, len(arr)):
  for j in range(i, 0, -1): #i부터 1까지 1씩 감소하며 반복
    if arr[j] < arr[j-1]:
      arr[j], arr[j-1] = arr[j-1], arr[j]
    else: #자기보다 작은 데이터를 만나면 그 위치에서 멈춤
      break
print(arr)

#1씩 감소하면서 자기 앞에꺼랑 비교해서 자리바꿈
#자기보다 더 작은 데이터를 만나면 멈춤
#데이터를 현재위치에서부터 앞으로 한칸씩 이동하면서 현재가 작으면 자리바꿔주고, 더 작은거 만나면 그자리에멈춤
